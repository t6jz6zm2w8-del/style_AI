import base64, io, json, os, signal, subprocess, threading, time, re
import numpy as np
from flask import Flask, request, render_template_string
from tensorflow.keras.models import load_model
from tensorflow.keras.preprocessing.image import img_to_array, load_img

# --- 0. Освобождаем порт 5000, если занят ---
os.system("fuser -k 5000/tcp 2>/dev/null")
time.sleep(1)


model = load_model('/content/style_classifier_improved.h5')
with open('/content/class_names.txt', 'r') as f:
    class_names = json.load(f)
print("✅ Модель загружена. Классы:", class_names)

HTML_PAGE = '''
<!doctype html>
<html lang="ru">
<head>
  <meta charset="utf-8">
  <title>Какой это стиль?</title>
  <style>
    body { font-family: Arial, sans-serif; max-width: 700px; margin: 2rem auto; padding: 1rem; }
    h1 { color: #3f51b5; }
    .upload-btn { background: #3f51b5; color: white; padding: 0.7em 1.5em; border: none; border-radius: 6px; cursor: pointer; }
    .result { margin-top: 1.5em; padding: 1em; background: #e8eaf6; border-radius: 10px; }
    img { margin-top: 1em; max-width: 300px; border-radius: 8px; box-shadow: 0 2px 8px rgba(0,0,0,0.2); }
  </style>
</head>
<body>
  <h1>🎨 Определитель стиля живописи</h1>
  <form method="post" enctype="multipart/form-data">
    <input type="file" name="file" accept="image/*" required>
    <br><br>
    <input class="upload-btn" type="submit" value="Распознать стиль">
  </form>
  {% if result %}
  <div class="result">
    <h2>Результат: {{ result }}</h2>
    <img src="{{ image_data }}" alt="Загруженное изображение">
  </div>
  {% endif %}
</body>
</html>
'''

app = Flask(__name__)

@app.route('/', methods=['GET', 'POST'])
def index():
    result = None
    image_data = None
    if request.method == 'POST':
        file = request.files['file']
        if file:
            img_bytes = file.read()
            encoded_img = base64.b64encode(img_bytes).decode('utf-8')
            image_data = f"data:image/jpeg;base64,{encoded_img}"

            file.seek(0)
            img = load_img(io.BytesIO(img_bytes), target_size=(224, 224))
            x = img_to_array(img) / 255.0
            x = np.expand_dims(x, axis=0)

            predictions = model.predict(x, verbose=0)
            class_idx = np.argmax(predictions[0])
            result = class_names[class_idx]

    return render_template_string(HTML_PAGE, result=result, image_data=image_data)

# --- 3. Запускаем Flask в отдельном потоке на порту 5000 ---
def run_flask():
    app.run(host='0.0.0.0', port=5000, debug=False, use_reloader=False)

thread = threading.Thread(target=run_flask)
thread.daemon = True
thread.start()
time.sleep(2)  # даём серверу запуститься

# --- 4. Скачиваем cloudflared (с удалением старого) ---
print("⏳ Устанавливаем Cloudflare Tunnel...")
os.system("rm -f /tmp/cloudflared")  # удаляем предыдущий
!wget -q -O /tmp/cloudflared https://github.com/cloudflare/cloudflared/releases/latest/download/cloudflared-linux-amd64
!chmod +x /tmp/cloudflared

# --- 5. Запускаем туннель и вытаскиваем ссылку ---
process = subprocess.Popen(
    ["/tmp/cloudflared", "tunnel", "--url", "http://localhost:5000"],
    stdout=subprocess.PIPE, stderr=subprocess.STDOUT, text=True
)

print("🔍 Ожидаем публичную ссылку...")
url = None
for line in iter(process.stdout.readline, ''):
    # Ищем строку с trycloudflare.com
    match = re.search(r'https://[^ ]+\.trycloudflare\.com', line)
    if match:
        url = match.group(0)
        break
    # Иногда ссылка выводится прямо в начале
    if 'trycloudflare.com' in line:
        parts = line.split()
        for p in parts:
            if p.startswith('https://') and 'trycloudflare.com' in p:
                url = p
                break
    if url:
        break

if url:
    print("\n" + "="*50)
    print("🚀 ГОТОВО! Откройте эту ссылку в браузере:")
    print(url)
    print("="*50)
    print("Сервер работает, пока эта ячейка выполняется.")
else:
    print("\n❌ Не удалось получить ссылку автоматически.")
    print("Попробуйте запустить туннель вручную в новой ячейке:")
    print("!/tmp/cloudflared tunnel --url http://localhost:5000")
